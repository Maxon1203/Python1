class Person:
 
    def __init__(self, name):
        self.__name = name   # –∏–º—è —á–µ–ª–æ–≤–µ–∫–∞
 
    # get –ø–æ–ª—É—á–∞–µ–º –∏–º—è —á–µ–ª–æ–≤–µ–∫–∞ 
    @property
    def Name(self):
        return self.__name
 
    def do_nothing(self):
        print(f"{self.Name} –Ω–∏—á–µ–≥–æ –Ω–µ –¥–µ–ª–∞–µ—Ç")
 
 
#  –∫–ª–∞—Å—Å —Ä–∞–±–æ—Ç–Ω–∏–∫–∞

    def work(self):
        print(f"{self.Name} —Ä–∞–±–æ—Ç–∞–µ—Ç")
 
 
#  –∫–ª–∞—Å—Å —Å—Ç—É–¥–µ–Ω—Ç–∞
class Student(Person):
 
    def study(self):
        print(f"{self.Name} —Å—Ç—É–¥–µ–Ω—Ç")
 
#  –ø—Ä–æ–≤–µ—Ä—è–µ–º –∫ –∫–∞–∫–æ–º—É –∫–ª–∞—Å—Å—É –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏—Ç —Ç–æ—Ç –∏–ª–∏ –∏–Ω–æ–π —á–µ–ª–æ–≤–µ–∫
def proverka(person):
    if isinstance(person, Student):
        person.study()
    elif isinstance(person, Employee):
        person.work()
    elif isinstance(person, Person):
        person.do_nothing()
